export const prioritizationLogic = {
  criteria: [
    { name: 'impact', weight: 0.5 },
    { name: 'urgency', weight: 0.5 },
  ],

    high: { impact: 8, urgency: 8 },
    medium: { impact: 5, urgency: 5 }
  }
};

### 2. Updated Feedback Model (`models/Feedback.js`)
```javascript
// models/Feedback.js
export class Feedback {
  constructor(data) {
    this.id = data.id;
    this.text = data.text;
    this.impact = data.impact || 0;
    this.urgency = data.urgency || 0;
    this.priority = data.priority || this.calculatePriority();
  }

  calculatePriority() {
    const { thresholds } = prioritizationLogic;
    if (this.impact >= thresholds.high.impact && this.urgency >= thresholds.high.urgency) {
      return 'high';
    } else if (this.impact >= thresholds.medium.impact || this.urgency >= thresholds.medium.urgency) {
      return 'medium';
    } else {
      return 'low';
    }
  }
}
```
