## 1. Acceptance Criteria
1. **Deterministic Ranking** – Given a list of feedback items, the algorithm returns the same ordered list on repeated runs with identical input and configuration.  
2. **Configurable Scoring** – The scoring weights for each criterion (e.g., impact, urgency, frequency) can be supplied via a JSON configuration file; changing the file updates the ranking without code changes.  
3. **Priority Levels** – Each feedback item is assigned a computed priority score and a derived priority label (`Critical`, `High`, `Medium`, `Low`).  
4. **Filter by Priority** – API endpoint `/feedback?priority=High` returns only items whose derived label matches the requested level, preserving the internal ranking order.  
5. **Graceful Degradation** – If a feedback item lacks a value for a configured criterion, the algorithm treats the missing value as zero and still produces a ranking.  
6. **Performance Bound** – Ranking 10 000 feedback items completes in ≤ 200 ms on a typical production node (v14+).  
7. **Audit Trail** – The algorithm outputs a `rankingLog` array containing `{id, score, criteriaBreakdown}` for each item, enabling downstream verification.

// prioritization.test.js
const { rankFeedback, loadConfig, filterByPriority } = require('./prioritization');

describe('Prioritization Module Unit Tests', () => {
  const defaultConfig = {
    weights: { impact: 0.5, urgency: 0.3, frequency: 0.2 },
    thresholds: { Critical: 80, High: 60, Medium: 40, Low: 0 },
  };

  const sampleFeedback = [
    { id: 'a', impact: 9, urgency: 8, frequency: 5 },
    { id: 'b', impact: 4, urgency: 9, frequency: 7 },
    { id: 'c', impact: 6, urgency: 5, frequency: 2 },
  ];

  test('rankFeedback returns deterministic order', () => {
    const first = rankFeedback(sampleFeedback, defaultConfig);
    const second = rankFeedback(sampleFeedback, defaultConfig);
    expect(first).toEqual(second);
  });

  test('score calculation respects weights', () => {
    const ranked = rankFeedback(sampleFeedback, defaultConfig);
    // manually compute expected score for item a
    const expectedScoreA = 9*0.5 + 8*0.3 + 5*0.2;
    expect(ranked[0].score).toBeCloseTo(expectedScoreA);
  });

  test('missing criterion defaults to zero', () => {
    const missing = [{ id: 'x', impact: 7 }]; // no urgency/frequency
    const ranked = rankFeedback(missing, defaultConfig);
    const expectedScore = 7*0.5 + 0*0.3 + 0*0.2;
    expect(ranked[0].score).toBeCloseTo(expectedScore);
  });

  test('priority label derived from thresholds', () => {
    const custom = [{ id: 'y', impact: 10, urgency: 10, frequency: 10 }]; // max score = 10
    const cfg = { ...defaultConfig, thresholds: { Critical: 9, High: 6, Medium: 3, Low: 0 } };
    const ranked = rankFeedback(custom, cfg);
    expect(ranked[0].priority).toBe('Critical');
  });

  test('loadConfig reads JSON and overrides defaults', () => {
    const cfgPath = '__tests__/fixtures/custom-config.json';
    const cfg = loadConfig(cfgPath);
    expect(cfg.weights.impact).toBe(0.7);
    expect(cfg.thresholds.High).toBe(70);
  });

  test('filterByPriority returns only matching items preserving order', () => {
    const ranked = rankFeedback(sampleFeedback, defaultConfig);
    const high = filterByPriority(ranked, 'High');
    expect(high.every(item => item.priority === 'High')).toBe(true);
    // ensure order unchanged
    expect(high.map(i => i.id)).toEqual(
      ranked.filter(i => i.priority === 'High').map(i => i.id)
    );
  });
});
