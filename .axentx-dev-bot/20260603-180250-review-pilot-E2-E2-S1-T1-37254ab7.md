## 2. Unit Tests (Jest style)

```js
// tests/prioritization.test.js
const { rankFeedback, loadConfig } = require('../prioritization');

describe('rankFeedback', () => {
  const sampleFeedback = [
    { id: 1, impact: 5, urgency: 4, recency: 2 },
    { id: 2, impact: 3, urgency: 5, recency: 1 },
  ];

  test('returns array of same length with score property', () => {
    const ranked = rankFeedback(sampleFeedback, { impact: 1, urgency: 1, recency: 1 });
    expect(ranked).toHaveLength(2);
    expect(ranked[0]).toHaveProperty('score');
  });

  test('higher impact yields higher score', () => {
    const ranked = rankFeedback(sampleFeedback, { impact: 2, urgency: 1, recency: 1 });
    expect(ranked[0].score).toBeGreaterThan(ranked[1].score);
  });

  test('config weights are applied correctly', () => {
    const config = { impact: 3, urgency: 1, recency: 0 };
    const ranked = rankFeedback(sampleFeedback, config);
    const expectedScore = 3 * 5 + 1 * 4 + 0 * 2; // 19
    expect(ranked[0].score).toBe(expectedScore);
  });

  test('throws error on missing fields', () => {
    const bad = [{ id: 3, impact: 4 }]; // missing urgency, recency
    expect(() => rankFeedback(bad, {})).toThrow('Missing required field');
  });
});

describe('loadConfig', () => {
  test('returns default weights if file missing', () => {
    const cfg = loadConfig('/nonexistent/path.json');
    expect(cfg).toEqual({ impact: 1, urgency: 1, recency: 1 });
  });

  test('parses JSON correctly', () => {
    const cfg = loadConfig('./test-config.json');
    expect(cfg.impact).toBe(2);
  });
});
```
