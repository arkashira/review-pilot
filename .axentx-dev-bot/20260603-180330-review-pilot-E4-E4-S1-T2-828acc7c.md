// /opt/axentx/review-pilot/notifications.js

const { Feedback, User, Notification } = require('../models');
const { Op } = require('sequelize');

class NotificationsService {
  /**
   * Initialize notification system
   */
  constructor() {
    this.statusUpdates = new Map();
  }

  /**
   * Track feedback status changes and notify users
   * @param {Object} feedback - Feedback object
   * @param {string} newStatus - New status value
   */
  async trackStatusChange(feedback, newStatus) {
    // Check if status has actually changed
    if (feedback.status === newStatus) return;
    
    // Update feedback status
    await feedback.update({ status: newStatus });
    
    // Generate notification
    const notification = {
      userId: feedback.userId,
      feedbackId: feedback.id,
      message: `Your feedback has been updated to: ${newStatus}`,
      timestamp: new Date(),
      type: 'status_update',
      status: newStatus
    };
    
    // Create and save notification
    await Notification.create(notification);
    
    // Trigger dashboard update
    this.updateDashboard(feedback.id, newStatus);
  }

  /**
   * Update dashboard with new status
   * @param {number} feedbackId - ID of feedback item
   * @param {string} status - New status
   */
  async updateDashboard(feedbackId, status) {
    // This would typically update the dashboard view
    console.log(`Dashboard updated: Feedback ${feedbackId} status changed to ${status}`);
    
    // In a real implementation, this would trigger a frontend update
    // e.g., socket.io event or API call to refresh the dashboard
  }

  /**
   * Get all notifications for a user
   * @param {number} userId - User ID
   * @returns {Promise<Array>} List of notifications
   */
  async getUserNotifications(userId) {
    return Notification.findAll({
      where: {
        userId,
        type: 'status_update'
      },
      order: [['timestamp', 'DESC']]
    });
  }

  /**
   * Mark notification as read
   * @param {number} notificationId - Notification ID
   */
  async markAsRead(notificationId) {
    await Notification.update(
      { read: true },
      { where: { id: notificationId } }
    );
  }
}

module.exports = new NotificationsService();
